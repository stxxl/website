185 template <class BIDType, class DiskAssignFunctor, class OutputIterator>
186 void block_manager::new_blocks_int(
187  const unsigned_type nblocks,
188  const DiskAssignFunctor& functor,
189  unsigned_type offset,
190  OutputIterator out)
191 {
192  typedef BIDType bid_type;
193  typedef BIDArray<bid_type::t_size> bid_array_type;
194 
195  simple_vector<int_type> bl(ndisks);
196  simple_vector<bid_array_type> disk_bids(ndisks);
197  simple_vector<file*> disk_ptrs(nblocks);
198 
199  bl.memzero();
200 
201  for (unsigned_type i = 0; i < nblocks; ++i)
202  {
203  unsigned_type disk = functor(offset + i);
204  disk_ptrs[i] = disk_files[disk];
205  bl[disk]++;
206  }
207 
208  for (unsigned_type i = 0; i < ndisks; ++i)
209  {
210  if (bl[i])
211  {
212  disk_bids[i].resize(bl[i]);
213  disk_allocators[i]->new_blocks(disk_bids[i]);
214  }
215  }
216 
217  bl.memzero();
218 
219  OutputIterator it = out;
220  for (unsigned_type i = 0; i != nblocks; ++it, ++i)
221  {
222  const int disk = disk_ptrs[i]->get_allocator_id();
223  bid_type bid(disk_ptrs[i], disk_bids[disk][bl[disk]++].offset);
224  *it = bid;
225  STXXL_VERBOSE_BLOCK_LIFE_CYCLE("BLC:new " << FMT_BID(bid));
226  }
227 
228 #if STXXL_MNG_COUNT_ALLOCATION
229  m_total_allocation += nblocks * BIDType::size;
230  m_current_allocation += nblocks * BIDType::size;
231  m_maximum_allocation = STXXL_MAX(m_maximum_allocation, m_current_allocation);
232 #endif // STXXL_MNG_COUNT_ALLOCATION
233 }

235 template <unsigned BlockSize>
236 void block_manager::delete_block(const BID<BlockSize>& bid)
237 {
238  // do not uncomment it
239  //assert(bid.storage->get_allocator_id() < config::get_instance()->disks_number());
240  if (!bid.is_managed())
241  return; // self managed disk
242  STXXL_VERBOSE_BLOCK_LIFE_CYCLE("BLC:delete " << FMT_BID(bid));
243  assert(bid.storage->get_allocator_id() >= 0);
244  disk_allocators[bid.storage->get_allocator_id()]->delete_block(bid);
245  disk_files[bid.storage->get_allocator_id()]->discard(bid.offset, bid.size);
246 
247 #if STXXL_MNG_COUNT_ALLOCATION
248  m_current_allocation -= BlockSize;
249 #endif // STXXL_MNG_COUNT_ALLOCATION
250 }
