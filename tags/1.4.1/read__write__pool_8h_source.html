92  // WRITE POOL METHODS
93 
94  //! Passes a block to the pool for writing.
95  //! \param block block to write. Ownership of the block goes to the pool.
96  //! \c block must be allocated dynamically with using \c new .
97  //! \param bid location, where to write
98  //! \warning \c block must be allocated dynamically with using \c new .
99  //! \return request object of the write operation
100  request_ptr write(block_type*& block, bid_type bid)
101  {
102  request_ptr result = w_pool->write(block, bid);
103 
104  // if there is a copy of this block in the prefetch pool,
105  // it is now a stale copy, so invalidate it and re-hint the block
106  if (p_pool->invalidate(bid))
107  p_pool->hint(bid, *w_pool);
108 
109  return result;
110  }

126  //! Gives a hint for prefetching a block.
127  //! \param bid address of a block to be prefetched
128  //! \return \c true if there was a free block to do prefetch and prefetching
129  //! was scheduled, \c false otherwise
130  //! \note If there are no free blocks available (all blocks
131  //! are already in reading or read but not retrieved by user calling \c read
132  //! method) calling \c hint function has no effect
133  bool hint(bid_type bid)

136  }

143  //! Reads block. If this block is cached block is not read but passed from the cache.

145  //! ownership goes to the pool and block from cache is returned in \c block value.
146  //! \param bid address of the block
147  //! \warning \c block parameter must be allocated dynamically using \c new .
148  //! \return request pointer object of read operation
